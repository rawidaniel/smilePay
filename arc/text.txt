version: "3"
services:
  sentry:
    image: getsentry/sentry:latest
    ports:
      - "9000:9000"
    volumes:
      - sentry-data:/var/lib/sentry/data
      - sentry-postgres:/var/lib/postgresql/data
    environment:
      - SENTRY_SECRET_KEY=<YOUR_SECRET_KEY>
      - SENTRY_POSTGRES_HOST=sentry-postgres
      - SENTRY_DB_USER=sentry
      - SENTRY_DB_PASSWORD=<YOUR_PASSWORD>
      - SENTRY_EMAIL_HOST=<YOUR_EMAIL_HOST>
      - SENTRY_EMAIL_PORT=<YOUR_EMAIL_PORT>
      - SENTRY_EMAIL_USER=<YOUR_EMAIL_USERNAME>
      - SENTRY_EMAIL_PASSWORD=<YOUR_EMAIL_PASSWORD>
      - SENTRY_EMAIL_USE_TLS=true
      - SENTRY_EMAIL_FROM_ADDRESS=<YOUR_EMAIL_FROM_ADDRESS>
      - SENTRY_EMAIL_SUBJECT_PREFIX=<YOUR_EMAIL_SUBJECT_PREFIX>
    depends_on:
      - sentry-postgres

  sentry-postgres:
    image: postgres:13
    volumes:
      - sentry-postgres:/var/lib/postgresql/data
    environment:
      - POSTGRES_PASSWORD=<YOUR_PASSWORD>
      - POSTGRES_USER=sentry
      - POSTGRES_DB=sentry

volumes:
  sentry-data:
  sentry-postgres:
